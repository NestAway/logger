  def start
    status = -1
    begin
      log( SEV_INFO, "Start of #{ @appName }." )
      status = run
    rescue
      log( SEV_FATAL, "Detected an exception. Stopping ... #{$!} (#{$!.type})\n" << $@.join( "\n" ))
    ensure
      log( SEV_INFO, "End of #{ @appName }. (status: #{ @status.to_s })" )
    end
    status
  end
